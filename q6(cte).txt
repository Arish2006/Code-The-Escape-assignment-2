int Solution::searchMatrix(vector<vector<int> > &A, int B) {
    int n=A.size();
    int m=A[0].size();
    vector<int> v(n);
    for(int i=0;i<n;i++){
        v[i]=A[i][0];
    }
    int row=upper_bound(v.begin(),v.end(),B)-v.begin()-1;
    return (row>=0?binary_search(A[row].begin(),A[row].end(),B):0);
}
