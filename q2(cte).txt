/**
 * @input A : Read only ( DON'T MODIFY ) Integer array
 * @input n1 : Integer array's ( A ) length
 * 
 * @Output Integer
 */
int findMin(const int* A, int n1) {
    int left=0,right=n1-1;
    while(left<right){
        int mid=left+(right-left)/2;
        if(A[mid]>A[right]){
            left=mid+1;
        }
        else{
            right=mid;
        }
        
    }
    return A[left];
}
