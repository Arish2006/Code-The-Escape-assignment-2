int Solution::solve(vector<int> &A, int B) {
    sort(A.begin(),A.end());
    int n1=A.size();
    int sum[n1];
    sum[n1-1]=A[n1-1];
    for(int i=n1-2;i>=0;i--){
        sum[i]=sum[i+1]+A[i];
    }
    int ans=0;
    for(int i=A[n1-1]-1;i>=0;i--){
        int j=upper_bound(A.begin(),A.end(),i)-A.begin();
        if(sum[j]-(n1-j)*i>=B){
            return i;
        }
    }
}
